Arduino Input Output Code


    // Starting code from in class example.

    // A0 one of Arduino's analog input pins to read a voltage btw 0V and 5V
    const int analogInPin = A0;
    // set up pins (PWM)
    const int green = 6;
    const int blue = 5;
    const int yellow = 3;
    void setup() {
    // set up pins as output:
    pinMode(green, OUTPUT);
    pinMode(blue, OUTPUT);
    pinMode(yellow, OUTPUT);
    // start serial monitor for printing data values:
    Serial.begin(9600);

    }

    // storing analog reading from pin A0, 0 - 990(measured values):
    int sensorValue;
    // storing brightness value (0-255);
    int mappedValue;

    void loop() {
    // put your main code here, to run repeatedly:
    // reads voltage, gives number 0 - 990
    sensorValue = analogRead(analogInPin);
    // print in serial monitor
    Serial.print("Sensor value: ");
    Serial.println(sensorValue);

    // scales sensorVaue to 0 - 255
    mappedValue = map(sensorValue, 0, 990, 0, 255);
    
    // sends PWM signals to LED pins
    // if the sensorValue is greater than 200(atleast a certain level of brightness)
    if(sensorValue > 200){
        // Print blurb to serial monitor
        Serial.println("Light detected â€” LEDs on!");
        // LEDs fade togther based on photoreceptor
        analogWrite(green, mappedValue);
        analogWrite(blue, mappedValue);
        analogWrite(yellow, mappedValue);
    }
    // if the sensorValue is less than 200
    else {
        // Print blurb to serial monitor
        Serial.println("It's dark - LEDs off!");
        // turn LEDs off
        digitalWrite(green, LOW);
        digitalWrite(blue, LOW);
        digitalWrite(yellow, LOW);
    }

    // delay to make it easier to read monitor
    delay(500);

    }
